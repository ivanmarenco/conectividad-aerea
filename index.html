async function apiGet(endpoint) {

  const sep = endpoint.includes('?') ? '&' : '?';

  const r = await fetch(`https://airlabs.co/api/v9/${endpoint}${sep}api_key=${state.apiKey}`);

  const d = await r.json();

  if (d.error) throw new Error(d.error.message || JSON.stringify(d.error));

  return d;

}

async function findAirport(city) {

  const key = city.trim().toLowerCase();

  

  // Verificar si es un código IATA directo (3 letras mayúsculas)

  if (/^[A-Z]{3}$/.test(city.trim().toUpperCase())) {

    const iata = city.trim().toUpperCase();

    return { iata_code: iata, name: iata, city: iata, country_code: '' };

  }

  

  // Buscar en mapa local

  const localIata = CITY_TO_IATA[key];

  if (localIata) {

    const iata = Array.isArray(localIata) ? localIata[0] : localIata;

    return { iata_code: iata, name: city, city: city, country_code: '' };

  }

  

  // Intentar con suggest API

  try {

    const d = await apiGet(`suggest?q=${encodeURIComponent(city)}`);

    const airports = d.response?.airports || [];

    if (airports.length) return airports[0];

  } catch(e) { /* fallback */ }

  

  throw new Error(`No se encontró aeropuerto para "${city}". Probá con el código IATA (ej: EZE, MAD, MIA)`);

}


// MÚLTIPLES LLAMADAS por aerolínea — evita el límite de 50 resultados

async function fetchRoutes(iata) {

  // Paso 1: obtener todas las aerolíneas que operan desde este aeropuerto

  showMsg('info', 'Obteniendo aerolíneas...');

  const firstCall = await apiGet(`routes?dep_iata=${iata}&limit=50&offset=0`);

  const firstBatch = firstCall.response || [];


  // Recolectar aerolíneas únicas de la primera página

  const airlineCodes = new Set();

  firstBatch.forEach(r => { if (r.airline_iata && !r.cs_airline_iata) airlineCodes.add(r.airline_iata); });


  // Si no hay más páginas, retornar directo

  if (!firstCall.request?.has_more) return firstBatch;


  // Paso 2: para cada aerolínea, traer sus rutas específicas desde este aeropuerto

  showMsg('info', `Cargando rutas de ${airlineCodes.size} aerolíneas...`);

  const allRoutes = [...firstBatch];

  const seen = new Set(firstBatch.map(r => r.flight_iata || r.flight_number));


  // También paginar la primera llamada por si hay más aerolíneas

  let offset = 50, hasMore = firstCall.request?.has_more;

  while (hasMore) {

    const d = await apiGet(`routes?dep_iata=${iata}&limit=50&offset=${offset}`);

    const batch = d.response || [];

    batch.forEach(r => {

      if (r.airline_iata && !r.cs_airline_iata) airlineCodes.add(r.airline_iata);

      const key = r.flight_iata || r.flight_number;

      if (!seen.has(key)) { seen.add(key); allRoutes.push(r); }

    });

    hasMore = d.request?.has_more === true && batch.length === 50;

    offset += 50;

    if (offset > 500) break;

  }


  // Paso 3: llamada por cada aerolínea para asegurar cobertura completa

  const airlineList = Array.from(airlineCodes);

  for (let i = 0; i < airlineList.length; i++) {

    const airline = airlineList[i];

    showMsg('info', `Cargando ${airline} (${i+1}/${airlineList.length})...`);

    try {

      let off = 0, more = true;

      while (more) {

        const d = await apiGet(`routes?dep_iata=${iata}&airline_iata=${airline}&limit=50&offset=${off}`);

        const batch = d.response || [];

        batch.forEach(r => {

          if (r.cs_airline_iata) return; // ignorar codeshares

          const key = (r.flight_iata || r.flight_number || '') + r.arr_iata;

          if (!seen.has(key)) { seen.add(key); allRoutes.push(r); }

        });

        more = d.request?.has_more === true && batch.length === 50;

        off += 50;

        if (off > 200) break;

      }

    } catch(e) { /* continuar con la siguiente aerolínea */ }

  }


  return allRoutes;

}


// ── PROCESS ─────────────────────────────────────────────────────────────────

function processRoutes(rawRoutes, originIata, domesticIATAs) {

  // Agrupar por arr_iata, deduplicar codeshares

  const byFlight = {};

  rawRoutes.forEach(r => {

    const arr = r.arr_iata;

    if (!arr || arr === originIata) return;

    if (r.cs_airline_iata) return; // ignorar codeshares

    const fkey = arr + '_' + (r.flight_iata || r.flight_number || Math.random());

    if (!byFlight[fkey]) byFlight[fkey] = { arr_iata: arr, airline: r.airline_iata, days: new Set() };

    (r.days || []).forEach(d => byFlight[fkey].days.add(d));

  });


  const grouped = {};

  Object.values(byFlight).forEach(f => {

    const arr = f.arr_iata;

    if (!grouped[arr]) grouped[arr] = { arr_iata: arr, airlines: new Set(), weekly: 0, domestic: domesticIATAs.has(arr) };

    grouped[arr].weekly += f.days.size;

    if (f.airline) grouped[arr].airlines.add(f.airline);

  });


  return Object.values(grouped)

    .map(g => ({ ...g, airlines: Array.from(g.airlines) }))

    .sort((a,b) => b.weekly - a.weekly);

}


// ── SEARCH ─────────────────────────────────────────────────────────────────

async function search() {

  const city = document.getElementById('cityInput').value.trim();

  if (!city) return;

  if (!state.apiKey) { toggleConfig(); showMsg('error','Configurá tu API key primero.'); return; }

  setLoading(true);

  document.getElementById('errorBar').classList.remove('show');


  try {

    const airport = await findAirport(city);

    showMsg('info', `Cargando rutas desde ${airport.name || airport.city} (${airport.iata_code})...`);


    // Si no tenemos country_code, lo obtenemos de la DB de aeropuertos

    if (!airport.country_code) {

      try {

        const d = await apiGet(`airports?iata_code=${airport.iata_code}`);

        const apt = (d.response || [])[0];

        if (apt) {

          airport.country_code = apt.country_code;

          airport.name = apt.name || airport.name;

          airport.city = apt.city || airport.city;

        }

      } catch(e) {}

    }


    const rawRoutes = await fetchRoutes(airport.iata_code);


    // Aeropuertos domésticos

    let domesticIATAs = new Set();

    try {

      if (airport.country_code) {

        const d = await apiGet(`airports?country_code=${airport.country_code}`);

        (d.response || []).forEach(a => domesticIATAs.add(a.iata_code));

      }

    } catch(e) {}


    const routes = processRoutes(rawRoutes, airport.iata_code, domesticIATAs);


    state.airport = airport;

    state.routes = routes;

    state.lastUpdated = new Date();


    document.getElementById('infoBar').classList.remove('show');

    render();

    document.getElementById('updateBtn').style.display = 'block';

  } catch(e) {

    showMsg('error', 'Error: ' + e.message);

  } finally {

    setLoading(false);

  }

}


async function updateData() {

  if (!state.airport) return;

  const btn = document.getElementById('updateBtn');

  btn.textContent = '⟳ Actualizando...'; btn.disabled = true;

  try {

    const rawRoutes = await fetchRoutes(state.airport.iata_code);

    const domMap = {};

    state.routes.forEach(r => { domMap[r.arr_iata] = r.domestic; });

    const domesticIATAs = new Set(Object.entries(domMap).filter(([,v])=>v).map(([k])=>k));

    state.routes = processRoutes(rawRoutes, state.airport.iata_code, domesticIATAs);

    state.lastUpdated = new Date();

    render();

  } catch(e) { showMsg('error','Error al actualizar: ' + e.message); }

  finally { btn.textContent = '⟳ Actualizar'; btn.disabled = false; }

}


// ── RENDER ─────────────────────────────────────────────────────────────────

function render() {

  document.getElementById('emptyState').style.display = 'none';

  document.getElementById('mainContent').classList.add('show');


  const { airport, routes, lastUpdated } = state;

  const domRoutes = routes.filter(r => r.domestic);

  const intlRoutes = routes.filter(r => !r.domestic);

  const totalW = routes.reduce((s,r) => s+(r.weekly||0), 0);

  const domW = domRoutes.reduce((s,r) => s+(r.weekly||0), 0);

  const intlW = intlRoutes.reduce((s,r) => s+(r.weekly||0), 0);


  // Airline summary

  const airMap = {};

  routes.forEach(r => {

    (r.airlines||[]).forEach(a => {

      if (!airMap[a]) airMap[a] = {code:a,routes:0,domestic:0,international:0,weekly:0};

      airMap[a].routes++; airMap[a].weekly += r.weekly||0;

      if (r.domestic) airMap[a].domestic++; else airMap[a].international++;

    });

  });

  state.airlines = Object.values(airMap).sort((a,b) => b.weekly - a.weekly);


  // Header

  document.getElementById('destCity').textContent = airport.city || airport.name;

  document.getElementById('destSub').textContent = `${airport.name} · ${airport.iata_code}`;

  const tag = document.getElementById('updatedTag');

  tag.style.display = 'block';

  tag.textContent = `Al ${lastUpdated.toLocaleDateString('es-AR')} ${lastUpdated.toLocaleTimeString('es-AR',{hour:'2-digit',minute:'2-digit'})}`;


  // Stats

  document.getElementById('statsGrid').innerHTML = [

    {v:totalW.toLocaleString(), l:'Vuelos / semana', s:'total', c:'#fff'},

    {v:routes.length, l:'Destinos directos', s:`${domRoutes.length} dom · ${intlRoutes.length} intl`, c:'#bfdbfe'},

    {v:domW.toLocaleString(), l:'Vuelos domésticos', s:`${domRoutes.length} destinos`, c:'#fdba74'},

    {v:intlW.toLocaleString(), l:'Vuelos internacionales', s:`${intlRoutes.length} destinos`, c:'#86efac'},

    {v:state.airlines.length, l:'Aerolíneas', s:'operadoras', c:'#c4b5fd'},

  ].map(s => `

    <div class="stat-box">

      <div class="val" style="color:${s.c}">${s.v}</div>

      <div class="lbl">${s.l}</div>

      <div class="sub2">${s.s}</div>

    </div>`).join('');


  renderMap();

  renderCities(domRoutes, intlRoutes);

  renderAirlines();

}

async function renderMap() {

  const svg = d3.select('#map-svg');

  svg.selectAll('*').remove();

  const W = document.getElementById('panel-map').clientWidth - 48;

  const H = Math.round(W * 0.5);

  svg.attr('width', W).attr('height', H);


  try {

    const world = await fetch('https://cdn.jsdelivr.net/npm/world-atlas@2/countries-110m.json').then(r=>r.json());

    const proj = d3.geoNaturalEarth1().scale(W/6.2).translate([W/2,H/2]);

    const path = d3.geoPath().projection(proj);


    svg.append('rect').attr('width',W).attr('height',H).attr('fill','#f0f9ff');

    svg.append('path').datum({type:'Sphere'}).attr('d',path).attr('fill','#dbeafe');

    svg.append('path').datum(d3.geoGraticule()()).attr('d',path).attr('fill','none').attr('stroke','#bfdbfe').attr('stroke-width',.4);

    const ctrs = topojson.feature(world, world.objects.countries);

    svg.selectAll('.land').data(ctrs.features).enter()

      .append('path').attr('class','land').attr('d',path)

      .attr('fill','#e2e8f0').attr('stroke','#cbd5e1').attr('stroke-width',.4);


    const airport = state.airport;

    const orig = COORDS[airport.iata_code] || [airport.lng, airport.lat];

    const maxW = Math.max(...state.routes.map(r=>r.weekly||1));


    // Arcos más finos primero

    [...state.routes].sort((a,b)=>(a.weekly||0)-(b.weekly||0)).forEach(r => {

      const dest = COORDS[r.arr_iata];

      if (!dest) return;

      svg.append('path')

        .datum({type:'LineString',coordinates:[orig,dest]})

        .attr('d',path).attr('fill','none')

        .attr('stroke',r.domestic?'#f97316':'#2563eb')

        .attr('stroke-width',Math.max(.7,((r.weekly||1)/maxW)*3.5))

        .attr('stroke-opacity',.65).attr('stroke-linecap','round');

    });


    // Puntos destino

    state.routes.forEach(r => {

      const dest = COORDS[r.arr_iata];

      if (!dest) return;

      const p = proj(dest);

      if (!p) return;

      svg.append('circle').attr('cx',p[0]).attr('cy',p[1]).attr('r',2.8)

        .attr('fill',r.domestic?'#f97316':'#2563eb')

        .attr('stroke','#fff').attr('stroke-width',1);

    });


    // Punto origen

    const op = proj(orig);

    if (op) {

      svg.append('circle').attr('cx',op[0]).attr('cy',op[1]).attr('r',14)

        .attr('fill','none').attr('stroke','#dc2626').attr('stroke-width',.8).attr('opacity',.2);

      svg.append('circle').attr('cx',op[0]).attr('cy',op[1]).attr('r',7)

        .attr('fill','none').attr('stroke','#dc2626').attr('stroke-width',1.5).attr('opacity',.5);

      svg.append('circle').attr('cx',op[0]).attr('cy',op[1]).attr('r',4)

        .attr('fill','#dc2626').attr('stroke','#fff').attr('stroke-width',2);

    }

  } catch(e) {

    svg.append('text').attr('x',W/2).attr('y',H/2).attr('text-anchor','middle').attr('fill','#94a3b8').text('No se pudo cargar el mapa');

  }

}
